\section{Discussion}
\label{sec:discussion}

\textbf{Strengths}
\begin{itemize}
    \item \textbf{Open-vocabulary reasoning:} Handles novel categories without retraining \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
    \item \textbf{Structured reasoning:} Scene graphs enable LLM reasoning about geometry \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
    \item \textbf{Robustness:} Resilient to segmentation errors and domain shifts \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
\end{itemize}

\textbf{Limitations}
\begin{itemize}
    \item \textbf{Perception dependency:} Relies on high-quality semantics \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
    \item \textbf{Latency:} LLM queries dominate runtime \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
    \item \textbf{Cost blindness:} High-level planning decoupled from execution costs \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
\end{itemize}

\textbf{Reproducibility}
\begin{itemize}
    \item \textbf{Code availability:} Project page available \citetip{moma_project}{MoMa-LLM Project Page}.
    \item \textbf{Missing details:} Real-world failure analysis limited \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
\end{itemize}

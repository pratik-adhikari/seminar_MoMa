\subsection{Experiments}
\label{sec:experiments}

\subsubsection{Experimental Setup}
\begin{itemize}
    \item \textbf{Simulator:} iGibson with 15 real-world scene scans \citetip{igibson}{iGibson (Li et al., 2021)}.
    \item \textbf{Robot (Sim):} Fetch mobile manipulator \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
    \item \textbf{Robot (Real):} Toyota HSR with RGB-D and LiDAR \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
    \item \textbf{Real Environment:} 4-room apartment (kitchen, living room, hallway, bathroom) with 54 object categories.
\end{itemize}

\subsubsection{Evaluation Metrics}
\begin{itemize}
    \item \textbf{Success Rate (SR):} Percentage of episodes where the target is found and \texttt{done()} is called.
    \item \textbf{SPL:} Success weighted by Path Length---measures navigation efficiency.
    \item \textbf{AUC-E:} Area Under the Efficiency Curve. \textit{Novel metric} integrating SR across all time budgets. High AUC-E indicates finding objects quickly, not just eventually \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
\end{itemize}

\textbf{Key Insights:}
\begin{itemize}
    \item MoMa-LLM achieves the highest SR (97.7\%), SPL (63.6), and AUC-E (87.2) \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
    \item Unstructured LLM (raw JSON) achieves only 86.3\% SR with 0.41 invalid actions/episode vs 0.19 for MoMa-LLM---structured prompting is essential.
    \item HIMOS achieves high SR (93.7\%) but poor efficiency (SPL 48.5), indicating brute-force exploration.
\end{itemize}

\textbf{Real-World Results}
\begin{itemize}
    \item Both MoMa-LLM and ESC achieved 80\% SR (8/10 episodes) \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
    \item MoMa-LLM traveled 17.9m vs 33.9m for ESC---nearly half the distance.
    \item MoMa-LLM required 2.2 interactions vs 3.5 for ESC.
\end{itemize}

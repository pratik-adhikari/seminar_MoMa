\subsection{Experiments}
\label{sec:experiments}

\subsubsection{Experimental Setup}
\begin{itemize}
    \item \textbf{Simulator:} iGibson (based on real-world scans) \citetip{igibson}{iGibson (Li et al., 2021)}.
    \item \textbf{Robot (Sim):} Fetch mobile manipulator \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
    \item \textbf{Robot (Real):} Toyota HSR \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
\end{itemize}

\subsubsection{Evaluation Metrics}
\begin{itemize}
    \item \textbf{Success Rate (SR):} Percentage of episodes where the target object is found.
    \item \textbf{SPL:} Success weighted by Path Length.
    \item \textbf{AUC-E:} Area Under the Efficiency Curve, a novel metric \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
\end{itemize}

\subsubsection{Results}

\textbf{Simulation}
\begin{itemize}
    \item \textbf{MoMa-LLM vs Baselines:} MoMa-LLM achieved SR 97.7\%, SPL 63.6, AUC-E 87.2 \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
    \item \textbf{Failure Modes:} Primary failures due to perception limitations \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
\end{itemize}

\textbf{Real-world Transfer}
\begin{itemize}
    \item \textbf{Transfer Capabilities:} 80\% success rate in real-world experiments \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
    \item \textbf{Latency:} Main constraints were LLM query latency \citetip{moma_llm}{MoMa-LLM (Honerkamp et al., 2024)}.
\end{itemize}

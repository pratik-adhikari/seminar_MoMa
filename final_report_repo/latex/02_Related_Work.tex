\section{Related Work}
\label{sec:related_work}

\subsection{3D Scene Graphs}
\begin{itemize}
    \item \textbf{Hydra:} Represents the state-of-the-art in real-time 3D Scene Graph construction, abstracting geometry into topology \citetip{hydra}{Hydra (Hughes et al., 2022)}.
    \item \textbf{Other:} Approaches like ConceptGraphs \citetip{conceptgraphs}{ConceptGraphs (Gu et al., 2024)} and VoroNav \citetip{voronav}{VoroNav (Wu et al., 2024)} investigate zero-shot perception inputs for task planning.
\end{itemize}

\subsection{LLMs for Planning}
\begin{itemize}
    \item \textbf{SayPlan:} Focuses on identifying subgraphs within large, known scene graphs for planning \citetip{sayplan}{SayPlan (Rana et al., 2023)}.
    \item \textbf{SayCan:} Grounds LLMs in robotic affordances but focuses on fully observed table-top settings \citetip{saycan}{SayCan (Ahn et al., 2022)}.
\end{itemize}

\subsection{Object Search}
\begin{itemize}
    \item \textbf{ESC:} A baseline for exploration with soft commonsense constraints, scoring frontiers based on object co-occurrences \citetip{esc}{ESC (Zhou et al., 2023)}.
    \item \textbf{HIMOS:} A hierarchical reinforcement learning approach for interactive search \citetip{himos}{HIMOS (Schmalstieg et al., 2023)}.
\end{itemize}
